# 5) Scatter plot math vs. reading colored by avg performance tier
def performance_bucket(score):
    if score < 50: return "Low"
    elif score < 70: return "Medium"
    else: return "High"
In[23]:
df["performance_category"] = df["avg_score"].apply(performance_bucket)

plt.figure(figsize=(8,6))
sns.scatterplot(x="math_score", y="reading_score", hue="performance_category", data=df, palette="Set1")
plt.title("Math vs Reading Score by Performance Category")
plt.show()
